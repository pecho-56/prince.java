#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, i, k, flag = 0;   // flag used for division or float detection
    char vari[15], typ[15], b[50], c;

    printf("Enter the number of variables: ");
    scanf("%d", &n);

    // Step 1: Read variable names and their types
    for (i = 0; i < n; i++)
    {
        printf("Enter the variable[%d]: ", i + 1);
        scanf(" %c", &vari[i]);

        printf("Enter the variable-type[%d] (float-f, int-i): ", i + 1);
        scanf(" %c", &typ[i]);
    }

    // Step 2: Read the expression
    printf("\nEnter the expression (end with $): ");
    i = 0;
    getchar(); // to clear buffer
    while ((c = getchar()) != '$')
    {
        b[i] = c;
        i++;
    }
    b[i] = '\0'; // terminate expression string
    k = i;

    // Step 3: Check if expression contains division
    for (i = 0; i < k; i++)
    {
        if (b[i] == '/')
        {
            flag = 1; // division present
            break;
        }
    }

    // Step 4: Type checking logic
    for (i = 0; i < n; i++)
    {
        if (b[0] == vari[i])  // check if first variable matches
        {
            if (flag == 1) // if division is used
            {
                if (typ[i] == 'f')
                    printf("\n✅ The datatype is correctly defined!\n");
                else
                    printf("\n❌ Identifier '%c' must be of float type (division used)!\n", vari[i]);
            }
            else
            {
                printf("\n✅ The datatype is correctly defined!\n");
            }
            break;
        }
    }

    return 0;
}
